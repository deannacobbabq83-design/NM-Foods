      const q = searchInput.value.toLowerCase().trim();
      Array.from(document.querySelectorAll('.recipe-card')).forEach(card => {
        const title = card.dataset.title.toLowerCase();
        const tags = card.dataset.tags.toLowerCase();
        const show = q === '' || title.includes(q) || tags.includes(q);
        card.style.display = show ? 'block' : 'none';
      });
    });

    // recipe data (could be moved to JSON or CMS)
    const recipes = {
      "Roasted Green Chile Stew": {

          "Sauté onion until translucent, add garlic and cook 1 minute.",
          "Return pork to pot, add green chiles and chicken broth. Bring to a boil, then reduce heat and simmer 50–60 minutes until pork is tender.",
          "Adjust seasoning with salt, pepper, and cumin. Serve hot with warm tortillas."
        ],
        notes: "Use roasted Hatch chiles when available. For milder stew, remove ribs and seeds from chiles."
      },

      "Sopaipillas": {
        id: "sopaipillas",
        title: "Sopaipillas",
        image: "sopaipillas.jpg",

          "Honey (for serving)"
        ],
        steps: [
          "Mix flour, baking powder, and salt. Cut in shortening until mixture is crumbly.",
          "Add warm water a little at a time and knead until smooth. Let rest 10 minutes.",
          "Roll dough out thin and cut into squares or triangles.",

          "Serve warm with honey."
        ],
        notes: "Don't overcrowd the pan when frying to allow them to puff up fully."
      },

      "Red Chile Pork": {
        id: "red-chile-pork",
        title: "Red Chile Pork",
        image: "red-chile-pork.jpg",
        servings: "6",
        time: "2 hrs (including simmer)",
        ingredients: [

    // Modal open/close & populate
    function openModal(recipeTitle){
      const r = recipes[recipeTitle];
      if(!r) return;
      const html = `
        <h2 style="color:var(--accent)">${r.title}</h2>
        <img src="${r.image}" alt="${r.title}" style="width:100%;max-height:300px;object-fit:cover;border-radius:8px;margin-top:8px">
        <div class="recipe-meta"><div>Serves: <strong>${r.servings}</strong></div><div>Time: <strong>${r.time}</strong></div></div>
        <div><strong>Ingredients</strong>
          <div class="ingredients"><ul>${r.ingredients.map(i=>`<li>${i}</li>`).join('')}</ul></div>
        </div>
        <div><strong>Steps</strong>
          <div class="steps"><ol>${r.steps.map(s=>`<li style="margin-bottom:8px">${s}</li>`).join('')}</ol></div>
        </div>
        <div style="color:var(--muted);margin-top:8px"><em>${r.notes}</em></div>
      `;
      document.getElementById('modalContent').innerHTML = html;
      document.getElementById('modal').classList.add('open');
      document.getElementById('modal').setAttribute('aria-hidden','false');
    }

    document.querySelectorAll('.open-recipe').forEach((btn, idx) => {
      btn.addEventListener('click', (e) => {
        const card = e.target.closest('.recipe-card');
        openModal(card.dataset.title);
      });
    });

    document.getElementById('closeModal').addEventListener('click', () => {
      document.getElementById('modal').classList.remove('open');
      document.getElementById('modal').setAttribute('aria-hidden','true');
    });

    // print button prints the open recipe or page
    document.getElementById('printBtn').addEventListener('click', () => {
      if(document.getElementById('modal').classList.contains('open')){
        window.print();
      } else {
        window.print();
      }
    });

    document.getElementById('viewAll').addEventListener('click', () => {
      window.location.hash = '#recipes';
    });

    // copy link function (demo)
    function copyLink(id){
      const url = window.location.href.split('#')[0] + '#' + id;
      navigator.clipboard?.writeText(url).then(()=>alert('Link copied!'));
    }

    // Optional: add deep-link support so #green-chile opens that recipe
    window.addEventListener('load', () => {
      const anchor = location.hash.replace('#','');
      if(anchor){
        // find recipe by id
        for(const k in recipes){
          if(recipes[k].id === anchor) openModal(k);
        }
      }
    });
